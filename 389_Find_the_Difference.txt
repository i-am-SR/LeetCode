class Solution {
    public char findTheDifference(String s, String t) {
        char freq[]=new char[26];
        int i;
        for(i=0;i<s.length();i++)
            freq[s.charAt(i)-'a']++;
        for(i=0;i<t.length();i++)
            freq[t.charAt(i)-'a']--;
        for(i=0;i<26;i++)
            if(freq[i]!=0) return (char)('a'+i);
        return ' ';
    }
}
